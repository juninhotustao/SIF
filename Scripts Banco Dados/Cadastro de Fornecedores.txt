CREATE TABLE FORNECEDORES(
	FOR_ID INT IDENTITY NOT NULL 
	CONSTRAINT PK_FOR_ID PRIMARY KEY (FOR_ID),

	FOR_TIPO_PES VARCHAR(1),
	FOR_CPF VARCHAR(15),
	FOR_CNPJ VARCHAR(15),	
	FOR_INSCRICAO VARCHAR(15),
	FOR_NOME VARCHAR(50),
	FOR_TELEFONE VARCHAR(15),
	FOR_TELEFONE2 VARCHAR(15),
	FOR_ENDERECO VARCHAR(50),
	FOR_END_NUMERO VARCHAR(10),
	FOR_END_COMPLEMENTO VARCHAR(30),
	FOR_BAIRRO VARCHAR(30),
	FOR_ESTADO VARCHAR(2),
	FOR_CIDADE VARCHAR(30),
	FOR_CEP VARCHAR(15)	
)

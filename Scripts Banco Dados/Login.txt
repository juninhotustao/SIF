IF (SELECT COUNT(*) FROM SYSOBJECTS WHERE ID = OBJECT_ID('USUARIOS') AND TYPE = 'U') = 0
BEGIN
	CREATE TABLE USUARIOS (
		USU_ID INT IDENTITY(1,1) PRIMARY KEY,
		USU_LOGIN VARCHAR(20)
		CONSTRAINT FK_USUARIOS_USU_LOGIN NOT NULL DEFAULT 'TESTE',
		USU_SENHA VARCHAR(20)
		CONSTRAINT FK_USUARIOS_USU_SENHA NOT NULL DEFAULT '1'
	)
END

CREATE TABLE CLIENTES(
	CLI_ID INT IDENTITY NOT NULL 
	CONSTRAINT PK_CLI_ID PRIMARY KEY (CLI_ID),

	CLI_DATA_CADASTRO DATETIME,
	CLI_TIPO_PES VARCHAR(1),
	CLI_CPF VARCHAR(15),
	CLI_CNPJ VARCHAR(15),
	CLI_INSCRICAO VARCHAR(15),
	CLI_RG VARCHAR(15), 
	CLI_EMISSAO_RG DATETIME,	
	CLI_NOME VARCHAR(50),
	CLI_NATURALIDADE VARCHAR(20),	
	CLI_NATURALIDADE_UF VARCHAR(20),
	CLI_NACIONALIDADE VARCHAR(20),	
	CLI_DATA_NASC DATETIME,
	CLI_TELEFONE VARCHAR(15),
	CLI_TELEFONE2 VARCHAR(15),
	CLI_ENDERECO VARCHAR(50),
	CLI_END_NUMERO VARCHAR(10),
	CLI_END_COMPLEMENTO VARCHAR(30),
	CLI_BAIRRO VARCHAR(30),
	CLI_ESTADO VARCHAR(2),
	CLI_CIDADE VARCHAR(30),
	CLI_CEP VARCHAR(15),
	CLI_ESTADO_CIVIL VARCHAR(20),
	CLI_SEXO VARCHAR(20)


)

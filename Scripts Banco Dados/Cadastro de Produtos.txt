CREATE TABLE PRODUTOS(
	PRO_ID INT IDENTITY NOT NULL 
	CONSTRAINT PK_PRO_ID PRIMARY KEY (PRO_ID),

	PRO_DATA_CADASTRO DATETIME NOT NULL,
	PRO_REFERENCIA VARCHAR(20),
	PRO_EAN VARCHAR(13),	
	PRO_DESCRICAO VARCHAR(60),
	PRO_UN_ID INT,
	GRP_ID INT, 
	PRO_ESTOQUE NUMERIC(9,3),
	PRO_PRECO_CUSTO NUMERIC(9,3),
	PRO_PRECO_CUSTO_REAL NUMERIC(9,3),
	PRO_PRECO_MEDIO_COMPRA NUMERIC(9,3),
	PRO_PRECO_VENDA NUMERIC(9,3),
	PRO_GAVETA VARCHAR(30),
	
	CONSTRAINT FK_PRO_UN_ID FOREIGN KEY (PRO_UN_ID) REFERENCES UNIDADE (UN_ID),
	CONSTRAINT FK_PRO_GRP_ID FOREIGN KEY (GRP_ID) REFERENCES GRUPO (GRP_ID)
	
)

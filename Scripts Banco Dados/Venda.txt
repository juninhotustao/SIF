CREATE TABLE VENDA(
	VEN_ID INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
	VEN_DATA DATETIME NOT NULL,
	CLI_ID INT,	
	VEN_TOTAL NUMERIC(15, 2) NOT NULL default 0,
	VEN_DESCONTO NUMERIC(15, 2) NOT NULL default 0,
	VEN_SITUACAO VARCHAR(1) NOT NULL default 'P',	
	VEN_OBSERVACAO VARCHAR(1000) NULL,	
	CONSTRAINT FK_VENDA_CLI_ID FOREIGN KEY (CLI_ID) REFERENCES CLIENTES (CLI_ID)	
)

CREATE TABLE ITEM_VENDA(
	ITV_ID INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
	VEN_ID INT NOT NULL,
	VEN_DATA DATETIME NOT NULL,
	PRO_ID INT NOT NULL,
	ITV_REFER VARCHAR(20) NOT NULL,
	ITV_QTDE NUMERIC(15, 3) NOT NULL,
	ITV_DESCONTO NUMERIC(15, 2) NULL,
	ITV_PRECOVENDA NUMERIC(15, 3) NOT NULL,
	ITV_VALORTOTAL NUMERIC(15, 2) NOT NULL,
	ITV_DATA DATETIME NULL,
	ITV_UN VARCHAR(10) NULL,	
	ITV_DESCRICAO VARCHAR(100) NULL,
	CONSTRAINT FK_ITEM_VENDA_PRO_ID FOREIGN KEY (PRO_ID) REFERENCES PRODUTOS (PRO_ID)	
)

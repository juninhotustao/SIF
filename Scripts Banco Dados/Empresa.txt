CREATE TABLE EMPRESA(
	EMP_ID INT IDENTITY NOT NULL 
	CONSTRAINT PK_EMP_ID PRIMARY KEY (EMP_ID),

	EMP_DATA_CADASTRO DATETIME,
	EMP_TIPO_PES VARCHAR(1),
	EMP_CPF VARCHAR(15),
	EMP_CNPJ VARCHAR(15),
	EMP_INSCRICAO VARCHAR(15),
	EMP_NOME VARCHAR(50),
	EMP_TELEFONE VARCHAR(15),
	EMP_TELEFONE2 VARCHAR(15),
	EMP_ENDERECO VARCHAR(50),
	EMP_END_NUMERO VARCHAR(10),
	EMP_END_COMPLEMENTO VARCHAR(30),
	EMP_BAIRRO VARCHAR(30),
	EMP_ESTADO VARCHAR(2),
	EMP_CIDADE VARCHAR(30),
	EMP_CEP VARCHAR(15),
	EMP_EMAIL VARCHAR(50)
)
